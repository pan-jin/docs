# 命名路由

```dart
void main() {
  runApp( Statelw());
}
//  Scafflod
class Statelw extends StatelessWidget{
  TextStyle textStyle = TextStyle(fontSize: 20);

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'StatelessWidget 及其组件',
      routes: {
        'home':(context) =>Home(),
        'product':(context)=> Product()
      },
      initialRoute: 'home', //设置默认路由页面
      onUnknownRoute: (RouteSettings settings)=>MaterialPageRoute(builder: (context)=>UnknowPage()), //404页面
      theme: ThemeData(
        primarySwatch: Colors.blue,
      ),
      home: AnonyMous());
  }

}
```
